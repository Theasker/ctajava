10 public class RoboCall05 {
11   public void phoneContacts(
12     List<Person> pl, Predicate<Person> pred){
13     
14     for(Person p:pl){
15       if (pred.test(p)){
16         roboCall(p);
17       }
18     }
19   }
20 
21   public void emailContacts(
22     List<Person> pl, Predicate<Person> pred){
23     
24     for(Person p:pl){
25       if (pred.test(p)){
26         roboEmail(p);
27       }
28     }
29   }
30 
31   public void mailContacts(
32     List<Person> pl, Predicate<Person> pred){
33     
34     for(Person p:pl){
35       if (pred.test(p)){
36         roboMail(p);
37       }
38     }
39   }  
40   
41   public void roboCall(Person p){
42     System.out.println("Calling " + p.getGivenName() 
43       + " " + p.getSurName() + " age " + p.getAge() 
44       + " at " + p.getPhone());
45   }
46   
47   public void roboEmail(Person p){
48     System.out.println("EMailing " + p.getGivenName() 
49       + " " + p.getSurName() + " age " + p.getAge() 
50       + " at " + p.getEmail());
51   }
52   
53   public void roboMail(Person p){
54     System.out.println("Mailing " + p.getGivenName() 
55       + " " + p.getSurName() + " age " + p.getAge() 
56       + " at " + p.getAddress() + ", " + p.getCity() 
57       + ", " + p.getState() + " " + p.getCode());
58   }
